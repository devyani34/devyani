vector<int> minAnd2ndMin(int a[], int n) {
    vector<int>arr;
   sort(a,a+n);
    for(int i=0;i<n;i++)
    {
        if(a[i]==a[i+1])
        continue;
        arr.push_back(a[i]);
    }  
    int c=arr.size();
    if(arr[0]==arr[c-1])
    {
        arr.emplace(arr.begin(),-1);
        return (arr);

// REMOVE VOWELS FROM STRING //
class Solution{
public:	
		
	string removeVowels(string S) 
	{
	    int l=S.length();
	  string str;
	    for(int i=0;i<l;i++)
	    {
	        
	        if(S.at(i)==97||S.at(i)==101||S.at(i)==105||S.at(i)==111||S.at(i)==117)
	        continue;
	        else
	        {
	            str.push_back(S[i]);
	            
	        }
	    }
	    
	    
	    return str;
	    
	}
};

//SORT COLORS//
class Solution {
public:
    void sortColors(vector<int>& nums) {
        int i;
        for(i=0;i<nums.size();i++)
        {
            for(int j=0;j<nums.size();j++)
            {
            if(nums.at(i)<nums.at(j))
            {
                swap(nums.at(i),nums.at(j));
            }
            }
        }
    }
};

int convertFive(int n) {
    int rem,num=0;

    {

        if(rem==0)
        rem=5;

    while(num>0)
    {
       int r=num%10;
        number=number*10+r;
        num=num/10;

int missingNumber(int A[], int N)
{
    int sum=0,sum1;
    
    for(int i=0;i<N-1;i++)
    {
        sum=sum+A[i];
    }
    sum1=N*(N+1)/2;
    int res=sum1-sum;
    return res;

}

	   {
	        if(arr[i]!=0)

//QUESTION 01//POWER OF 2
class Solution{
    public:
    // Function to check if given number n is a power of two.
    bool isPowerofTwo(long long n){
        if(n==1)
        return 1;
        if(n==0)
        return 0;
        while(n>2)
        {
            if(n%2==0)
            n=n/2;
            else
            return 0;
        }
        return 1;
     }
};

class Solution{
public:
    int primeSum(int N){
        // code here
        int sum=0,count=0;
        while(N!=0)
        {
            int rem=N%10;
            if(rem==2||rem==3||rem==5||rem==7)
          {
             count=1;
            sum=sum+rem;
        
          }
          N=N/10;
          
        }
        
        if(count==0)
        return 0;
         return sum;
        }
};

class Solution{
