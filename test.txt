#include<iostream>
using namespace std;
int main ()
{
    int i,j,n;
    cout<<"enter the number of the rows:";
    cin>>n;
    for(i=1;i<=n;i++)
    {
        for(j=1;j<=n-i;j++)
        {
            cout<<" ";
        }
        for(j=1;j<=i;j++)
        {
            cout<<"* ";
        }
        cout<<endl;
    }
    return 0;
}

vector<int> minAnd2ndMin(int a[], int n) {
    vector<int>arr;
   sort(a,a+n);
    for(int i=0;i<n;i++)
    {
        if(a[i]==a[i+1])
        continue;
        arr.push_back(a[i]);
    }  
    int c=arr.size();
    if(arr[0]==arr[c-1])
    {
        arr.emplace(arr.begin(),-1);
        return (arr);
    }
    
    return (arr);
    
    
}
